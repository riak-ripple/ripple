def inspect
  attribute_list = attributes_for_persistence.except("_type").map {|k,v| "#{k}=#{v.inspect}" }.join(' ')
  identifier = self.class.embeddable? ? "" : ":#{key || '[new]'}"
  "<#{self.class.name}#{identifier} #{attribute_list}>"
end
