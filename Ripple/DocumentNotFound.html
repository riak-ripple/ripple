def initialize(keys, found)
  if keys.empty?
    super(t("document_not_found.no_key"))
  elsif keys.one?
    super(t("document_not_found.one_key", :key => keys.first))
  else
    missing = keys - found.compact.map(&:key)
    super(t("document_not_found.many_keys", :keys => missing.join(', ')))
  end
end
